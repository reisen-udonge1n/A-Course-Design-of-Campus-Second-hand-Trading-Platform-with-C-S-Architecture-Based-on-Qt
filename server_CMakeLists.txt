cmake_minimum_required(VERSION 3.16)
project(chat_server LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)

find_package(Qt6 COMPONENTS Network QUIET)
if(NOT Qt6_FOUND)
    find_package(Qt5 5.15 REQUIRED COMPONENTS Network)
endif()

add_executable(chat_server
    server_main.cpp
    chatserver.h
    chatserver.cpp
)

if(Qt6_FOUND)
    target_link_libraries(chat_server PRIVATE Qt6::Network)
else()
    target_link_libraries(chat_server PRIVATE Qt5::Network)
endif()
